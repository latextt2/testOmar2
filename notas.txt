[cambiar de rama] -> git checkout <nombre de la rama>

[mostrar ramas en log] -> git log --graph

[mover a carpeta] -> cd <nombre>

[subir a github] -> git push origin master

[bajar de github] -> git pull origin master

[crear rama] -> git branch <nobre de la rama>

[agregar todos los archivos] -> git al

[ver status] -> git st

[crear commit] -> git cim "nombre"

Pasos:

git checkout omar -> [mover cosas] -> git st -> git al -> git st -> git cim -> git st -> git checkout master -> git pull origin master -> git checkout omar -> git merge master -> git st [corregir errores si hay] ->  git al -> git cim -> git checkout master -> git merge omar ->git push origin master
